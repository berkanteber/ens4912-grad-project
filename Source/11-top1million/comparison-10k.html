wself = {}
for i, p in enumerate(hws):
    if p in wself: wself[p]["h"] = i + 1
    else: wself[p] = { "h": i + 1, "g": None, "hp": None }
for i, p in enumerate(gws):
    if p in wself: wself[p]["g"] = i + 1
    else: wself[p] = { "g": i + 1, "h": None, "hp": None }
for i, p in enumerate(hpws):
