    data = np.array([hwwo[p]["w"] for p in common if hwwo[p]["comm"] == i])
    kde = scipy.stats.gaussian_kde(data)
    plt.plot(np.linspace(0, 1000000, 1000), kde(np.linspace(0, 1000000, 1000)), label='Community #'+str(i))
